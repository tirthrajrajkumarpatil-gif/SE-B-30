    <script>
        // PAGE SWITCHING
        function showTab(type) {
            document.getElementById('login-form').classList.toggle('hidden', type !== 'login');
            document.getElementById('signup-form').classList.toggle('hidden', type !== 'signup');
            document.getElementById('t-login').classList.toggle('active', type === 'login');
            document.getElementById('t-signup').classList.toggle('active', type === 'signup');
        }

        // AUTH LOGIC
        function handleAuth(type) {
            let name = type === 'login' 
                ? (document.getElementById('login-user').value || "User") 
                : (document.getElementById('signup-name').value || "New User");
            
            let mode = type === 'signup' ? document.getElementById('mode-select').value : "Standard";

            // Visual Transition
            const card = document.querySelector('.auth-card');
            card.style.transform = "scale(0.9)";
            card.style.opacity = "0";

            setTimeout(() => {
                document.getElementById('auth-page').classList.add('hidden');
                document.getElementById('dashboard-page').classList.remove('hidden');
                
                document.getElementById('display-name').innerText = name;
                document.getElementById('display-mode').innerText = "Mode: " + mode;
                document.getElementById('session-id').innerText = "SID-" + Math.random().toString(36).substring(7).toUpperCase();
                
                speak("Welcome back " + name + ". System initialized.");
            }, 400);
        }

        function logout() { location.reload(); }

        // ASSISTIVE TECH
        function speak(text) {
            if ('speechSynthesis' in window) {
                window.speechSynthesis.cancel(); // Stop current speech
                const utterance = new SpeechSynthesisUtterance(text);
                utterance.pitch = 1;
                utterance.rate = 0.9;
                window.speechSynthesis.speak(utterance);
            }
        }

        function triggerSOS() {
            speak("Emergency Protocol Initiated. Sending location to family.");
            alert("🚨 EMERGENCY ALERT SENT!");
        }

        window.onload = () => {
            console.log("UASS System Online");
        };
    </script>
